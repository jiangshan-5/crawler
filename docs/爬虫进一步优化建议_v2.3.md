#### 2. 异步并发爬取 ⭐⭐⭐⭐⭐
**问题**: 当前是单线程顺序爬取，速度慢

**优化方案**:
```python
# 使用 asyncio + aiohttp 实现异步爬取
import asyncio
import aiohttp

class AsyncCrawler:
    async def fetch_multiple(self, urls, max_concurrent=5):
        """并发爬取多个 URL"""
        semaphore = asyncio.Semaphore(max_concurrent)
        
        async with aiohttp.ClientSession() as session:
            tasks = [
                self._fetch_one(session, url, semaphore) 
                for url in urls
            ]
            return await asyncio.gather(*tasks)
    
    async def _fetch_one(self, session, url, semaphore):
        async with semaphore:
            async with session.get(url) as response:
                return await response.text()
```

**收益**:
- 速度提升 5-10 倍（取决于并发数）
- 更高效利用网络带宽
- 适合大规模爬取

**实现难度**: 中等

**注意事项**:
- 需要控制并发数，避免被封禁
- 需要处理异步异常
- GUI 需要支持异步更新

class DataDeduplicator:
    """数据去重器"""
    
    def __init__(self, cache_file='data/crawled_urls.json'):
        self.cache_file = cache_file
        self.crawled_urls = self._load_cache()
    
    def is_crawled(self, url):
        """检查 URL 是否已爬取"""
        url_hash = hashlib.md5(url.encode()).hexdigest()
        return url_hash in self.crawled_urls
    
    def mark_crawled(self, url, data_hash):
        """标记 URL 已爬取"""
        url_hash = hashlib.md5(url.encode()).hexdigest()
        self.crawled_urls[url_hash] = {
            'url': url,
            'data_hash': data_hash,
            'timestamp': datetime.now().isoformat()
        }
        self._save_cache()
    
    def _load_cache(self):
        """加载缓存"""
        if os.path.exists(self.cache_file):
            with open(self.cache_file, 'r') as f:
                return json.load(f)
        return {}
    
    def _save_cache(self):
        """保存缓存"""
        os.makedirs(os.path.dirname(self.cache_file), exist_ok=True)
        with open(self.cache_file, 'w') as f:
            json.dump(self.crawled_urls, f, indent=2)
```

**收益**:
- 避免重复爬取
- 支持增量更新
- 节省时间和带宽

**实现难度**: 简单

class ProxyPool:
    """代理池管理器"""
    
    def __init__(self, proxy_list=None):
        self.proxies = proxy_list or []
        self.current_index = 0
        self.failed_proxies = set()
    
    def get_proxy(self):
        """获取可用代理"""
        if not self.proxies:
            return None
        
        # 轮询选择代理
        for _ in range(len(self.proxies)):
            proxy = self.proxies[self.current_index]
            self.current_index = (self.current_index + 1) % len(self.proxies)
            
            if proxy not in self.failed_proxies:
                return proxy
        
        return None
    
    def mark_failed(self, proxy):
        """标记代理失败"""
        self.failed_proxies.add(proxy)
    
    def reset_failed(self):
        """重置失败代理（定期调用）"""
        self.failed_proxies.clear()

# 在爬虫中使用
class UniversalCrawlerV3:
    def __init__(self, proxy_pool=None):
        self.proxy_pool = proxy_pool
    
    def fetch_page(self, url):
        proxy = self.proxy_pool.get_proxy() if self.proxy_pool else None
        
        try:
            response = self.session.get(
                url, 
                proxies={'http': proxy, 'https': proxy} if proxy else None
            )
            return response
        except:
            if proxy:
                self.proxy_pool.mark_failed(proxy)
            raise
```

from queue import Queue
import threading

class TaskScheduler:
    """任务调度器"""
    
    def __init__(self, max_workers=3):
        self.task_queue = Queue()
        self.max_workers = max_workers
        self.workers = []
        self.running = False
    
    def add_task(self, task):
        """添加任务"""
        self.task_queue.put(task)
    
    def start(self):
        """启动调度器"""
        self.running = True
        for i in range(self.max_workers):
            worker = threading.Thread(target=self._worker, args=(i,))
            worker.daemon = True
            worker.start()
            self.workers.append(worker)
    
    def _worker(self, worker_id):
        """工作线程"""
        while self.running:
            try:
                task = self.task_queue.get(timeout=1)
                self._execute_task(task, worker_id)
                self.task_queue.task_done()
            except:
                continue
    
    def _execute_task(self, task, worker_id):
        """执行任务"""
        print(f"Worker {worker_id} 正在执行: {task['name']}")
        # 执行爬取逻辑
        pass
```

class ConfigManager:
    """配置管理器"""
    
    def save_config(self, name, config):
        """保存配置"""
        config_dir = 'configs'
        os.makedirs(config_dir, exist_ok=True)
        
        filepath = os.path.join(config_dir, f"{name}.json")
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(config, f, ensure_ascii=False, indent=2)
    
    def load_config(self, name):
        """加载配置"""
        filepath = os.path.join('configs', f"{name}.json")
        if os.path.exists(filepath):
            with open(filepath, 'r', encoding='utf-8') as f:
                return json.load(f)
        return None
    
    def list_configs(self):
        """列出所有配置"""
        config_dir = 'configs'
        if not os.path.exists(config_dir):
            return []
        
        return [
            f.replace('.json', '') 
            for f in os.listdir(config_dir) 
            if f.endswith('.json')
        ]
```

class DataExporter:
    """数据导出器"""
    
    def export_to_excel(self, data, filepath):
        """导出为 Excel"""
        import pandas as pd
        df = pd.DataFrame(data)
        df.to_excel(filepath, index=False, engine='openpyxl')
    
    def export_to_sqlite(self, data, db_path, table_name):
        """导出为 SQLite 数据库"""
        import sqlite3
        import pandas as pd
        
        df = pd.DataFrame(data)
        conn = sqlite3.connect(db_path)
        df.to_sql(table_name, conn, if_exists='append', index=False)
        conn.close()
    
    def export_to_markdown(self, data, filepath):
        """导出为 Markdown 表格"""
        if not data:
            return
        
        # 生成表头
        headers = list(data[0].keys())
        md = '| ' + ' | '.join(headers) + ' |\n'
        md += '| ' + ' | '.join(['---'] * len(headers)) + ' |\n'
        
        # 生成数据行
        for row in data:
            values = [str(row.get(h, '')) for h in headers]
            md += '| ' + ' | '.join(values) + ' |\n'
        
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(md)
```

class DataCleaner:
    """数据清洗器"""
    
    @staticmethod
    def clean_price(text):
        """清洗价格数据"""
        import re
        # 提取数字
        match = re.search(r'[\d,]+\.?\d*', text)
        if match:
            return float(match.group().replace(',', ''))
        return None
    
    @staticmethod
    def clean_date(text):
        """清洗日期数据"""
        from dateutil import parser
        try:
            return parser.parse(text).isoformat()
        except:
            return text
    
    @staticmethod
    def clean_url(url, base_url):
        """清洗 URL（转为绝对路径）"""
        from urllib.parse import urljoin
        return urljoin(base_url, url)
```
